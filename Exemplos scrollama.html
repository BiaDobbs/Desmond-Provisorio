JavaScript - Sticky Overlay
function handleStepEnter(response) {
  const stepNum = parseInt(response.element.dataset.step);
  
  // Remove estados anteriores
  document.querySelectorAll('.visual-element').forEach(el => {
    el.classList.remove('active');
  });
  
  // Ativa elemento específico
  document.querySelector(`[data-visual="${stepNum}"]`)?.classList.add('active');
}

2. Progress-Based Animation

Animações suaves baseadas no progresso do scroll. Perfeito para mostrar evolução, carregamento ou transformações graduais.

0%

Progresso Contínuo

Esta barra progride suavemente conforme você faz scroll. O progresso vai de 0% a 100% durante toda a duração deste step. Ideal para mostrar crescimento, evolução temporal ou qualquer métrica que muda gradualmente.

JavaScript - Progress Animation
function handleStepProgress(response) {
  const progress = response.progress; // 0 a 1
  const percentage = Math.round(progress * 100);
  
  document.querySelector('.progress-fill').style.width = `${percentage}%`;
  document.querySelector('.progress-text').textContent = `${percentage}%`;
}

3. Multi-Phase Animation

Divide uma animação em múltiplas fases dentro de um único step. Útil para sequências complexas ou storytelling em etapas.

🎭

Transformação em Fases

Este elemento passa por três fases distintas: Fase 1 (0-33%) - Aparece e cresce; Fase 2 (33-66%) - Rotaciona e muda cor; Fase 3 (66-100%) - Se transforma e se move. Cada fase tem sua própria animação dentro do mesmo step.

JavaScript - Multi-Phase Animation
function animateInPhases(progress) {
  const element = document.querySelector('.phase-element');
  
  if (progress < 0.33) {
    // Fase 1: Aparecer e crescer
    const phaseProgress = progress / 0.33;
    element.style.transform = `scale(${phaseProgress}) rotate(0deg)`;
    element.style.opacity = phaseProgress;
  } else if (progress < 0.66) {
    // Fase 2: Rotacionar
    const phaseProgress = (progress - 0.33) / 0.33;
    element.style.transform = `scale(1) rotate(${phaseProgress * 360}deg)`;
    element.style.background = `hsl(${phaseProgress * 240}, 70%, 50%)`;
  } else {
    // Fase 3: Mover e transformar
    const phaseProgress = (progress - 0.66) / 0.34;
    element.style.transform = `scale(${1 + phaseProgress * 0.5}) rotate(360deg) translateX(${phaseProgress * 100}px)`;
  }
}

JavaScript - Chart Animation
function animateChart(progress, stepIndex) {
  const bars = document.querySelectorAll('.bar');
  
  bars.forEach((bar, i) => {
    const value = parseInt(bar.dataset.value);
    const delay = i * 0.1; // Animação escalonada
    
    if (progress > delay) {
      const adjustedProgress = Math.min((progress - delay) / (1 - delay), 1);
      bar.style.height = `${value * adjustedProgress}%`;
    }
  });
}

JavaScript - Interactive Map
function handleStepEnter(response) {
  const stepData = response.element.dataset.step;
  
  // Remove highlights anteriores
  document.querySelectorAll('.map-region').forEach(region => {
    region.classList.remove('highlighted');
  });
  
  // Destaca região específica
  const regionMap = {
    'map1': 'north',
    'map2': 'south', 
    'map3': 'east',
    'map4': 'west'
  };
  
  const regionId = regionMap[stepData];
  if (regionId) {
    document.getElementById(`region-${regionId}`).classList.add('highlighted');
  }
}

JavaScript - Timeline Progress
function updateTimeline(stepData) {
  const yearMap = {
    'timeline1': 2020,
    'timeline2': 2021,
    'timeline3': 2022,
    'timeline4': 2023
  };
  
  const currentYear = yearMap[stepData];
  if (!currentYear) return;
  
  // Atualiza pontos da timeline
  document.querySelectorAll('.timeline-point').forEach(point => {
    const pointYear = parseInt(point.dataset.year);
    point.classList.toggle('active', pointYear <= currentYear);
  });
  
  // Atualiza progresso
  const progress = ((currentYear - 2020) / 3) * 100;
  document.getElementById('timeline-progress').style.height = `${progress}%`;
}
